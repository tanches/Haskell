module Ackermann where

-- функция принимает на вход 2 числа. Рекурсивное вычисление происходит до тех пор, пока первое входное число не станет = 0
ackermann :: Int -> Int -> Int
ackermann x y
    -- есть три реализации вычисления в зависимости от входных параметров
    |x == 0 = y + 1
    -- рекурсивное вычисление
    |(x > 0) && (y == 0) = ackermann (x - 1) 1
    -- рекурсивное вычисление
    |(x > 0) && (y > 0) = ackermann (x - 1) (ackermann x (y - 1))

main = putStrLn (show (ackermann 4 0))
